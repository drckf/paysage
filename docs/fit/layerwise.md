def compute_StateTAP(self, use_GD=True, init_lr=0.1, tol=1e-07, max_iters=50, ratchet=True, decrease_on_neg=0.9, mean_weight=0.9, mean_square_weight=0.999, seed=None, rescaled_weight_cache=None)
